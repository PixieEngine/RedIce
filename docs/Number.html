Number.prototype.approachRotation = function(target, maxDelta) {
  while (target > this + Math.PI) {
    target -= Math.TAU;
  }
  while (target < this - Math.PI) {
    target += Math.TAU;
  }
  return (target - this).clamp(-maxDelta, maxDelta) + this;
};

Number.prototype.constrainRotation = function() {
  var target;
  target = this;
  while (target > Math.PI) {

Number#primeFactors()

      

        
In number theory, integer factorization or prime factorization is the
breaking down of a composite number into smaller non-trivial divisors,
which when multiplied together equal the original integer.

        

          
Usage

          
Floors the number for purposes of factorization.




60.primeFactors()



=> [2, 2, 3, 5]



37.primeFactors()



=> [37]




        

        

          
Returns

          

            Type:
            Array
          

          
An array containing the factorization of this number.

        

        

          
Source

          Toggle
          
Number.prototype.primeFactors = function() {
  var factors, i, iSquared, n;
  factors = [];
  n = Math.floor(this);
  if (n === 0) return;
  if (n < 0) {
    factors.push(-1);
    n /= -1;
  }
  i = 2;
  iSquared = i * i;
  while (iSquared < n) {
    while ((n % i) === 0) {
      factors.push(i);
      n /= i;
    }
    i += 1;
    iSquared = i * i;
  }
  if (n !== 1) factors.push(n);
  return factors;
};

Number#times(iterator, context)

      

        
Calls iterator the specified number of times, passing in the number of the 
current iteration as a parameter: 0 on first call, 1 on the second call, etc. 

        

          
Usage

          

output = []



5.times (n) ->
  output.push(n)



output



=> [0, 1, 2, 3, 4]




        

        

          
Parameters

          
    
            
  • 
              iterator
               -
              The iterator takes a single parameter, the number of the current iteration.
            
    • 
            
  • 
              context
               -
              The optional context parameter specifies an object to treat as this in the iterator block.
            
    • 
          

        

        

          
Returns

          

            Type:
            Number
          

          
The number of times the iterator was called.

        

        

          
Source

          Toggle
          
Number.prototype.times = function(iterator, context) {
  var i;
  i = -1;
  while (++i < this) {
    iterator.call(context, i);
  }
  return i;
};
